  <script>
    const { jsPDF } = window.jspdf;
    const API_BASE = 'http://localhost:3000/api';
    let apiToken = null;

    let headersAtt = [], attendanceData = [];
    let headersUsr = [], usersData = [];
    let currentUser = null;

    // 1) تسجيل دخول المشرف
    async function adminLogin() {
      const code = document.getElementById('adminCode').value.trim();
      const pass = document.getElementById('adminPass').value.trim();
      try {
        const res = await fetch(`${API_BASE}/admin/login`, {
          method:'POST',
          headers:{ 'Content-Type':'application/json' },
          body: JSON.stringify({ code, password: pass })
        });
        if (!res.ok) throw '';
        const { token } = await res.json();
        apiToken = token;
        document.getElementById('adminUpload').classList.remove('hidden');
      } catch {
        alert('بيانات المسؤول خاطئة');
      }
    }

    // 2) رفع الملف
    document.getElementById('fileInput').addEventListener('change', async e => {
      const file = e.target.files[0]; if (!file) return;
      const form = new FormData();
      form.append('file', file);
      try {
        const res = await fetch(`${API_BASE}/upload`, {
          method:'POST',
          headers: { 'Authorization': `Bearer ${apiToken}` },
          body: form
        });
        if (!res.ok) throw '';
        alert('تم رفع وتخزين البيانات على الخادم');
        await fetchData();
      } catch {
        alert('فشل في رفع الملف');
      }
    });

    // 3) تحديث البيانات
    async function refreshData() {
      if (!apiToken) return alert('ليس مسموحاً');
      await fetchData();
      alert('تم تحديث البيانات');
    }

    // 4) جلب Users و Attendance
    async function fetchData() {
      const [uRes,aRes] = await Promise.all([
        fetch(`${API_BASE}/users`,      { headers:{ Authorization:`Bearer ${apiToken}` }}),
        fetch(`${API_BASE}/attendance`, { headers:{ Authorization:`Bearer ${apiToken}` }})
      ]);
      if (uRes.ok && aRes.ok) {
        ({ headers: headersUsr, data: usersData }      = await uRes.json());
        ({ headers: headersAtt, data: attendanceData } = await aRes.json());
      }
    }

    // 5) تسجيل دخول الموظف
    async function login() {
      const code = document.getElementById('codeInput').value.trim();
      const pass = document.getElementById('passwordInput').value.trim();
      if (!usersData.length) return alert('يرجى رفع ملف الإكسل أولاً');
      const idxC = headersUsr.indexOf('رقم الموظف');
      const idxP = headersUsr.indexOf('كلمة المرور');
      const idxName = headersAtt.indexOf('الاسم');
      const recUser = usersData.find(r=>String(r[idxC])===code && String(r[idxP])===pass);
      if (!recUser) return alert('بيانات الدخول خاطئة');

      currentUser = code;
      // أحتاج أول سجل للـ KPI
      const rec = attendanceData.find(r=> String(r[idxC])===code) || [];
      document.getElementById('welcomeMsg').textContent = `مرحباً ${rec[idxName]||''}`;
      document.getElementById('welcomeName').textContent = `(الكود: ${code})`;

      // أعمدة رقم 13 و14 و15 في Excel هي الفهرس 12,13,14
      document.getElementById('prodIncentivePercent').textContent = rec[12] ?? '--';
      document.getElementById('adminLeavesCounted').textContent   = rec[13] ?? '--';
      document.getElementById('adminLeavesRemaining').textContent = rec[14] ?? '--';

      document.getElementById('loginSection').classList.add('hidden');
      document.getElementById('records').classList.remove('hidden');
      renderRecords();
    }

    // 6) عرض السجلات
    function renderRecords(){
      const tbody = document.getElementById('attendanceBody');
      tbody.innerHTML = '';
      const idx = {
        code: headersAtt.indexOf('رقم الموظف'),
        date: headersAtt.indexOf('التاريخ'),
        status: headersAtt.indexOf('رمز الحالة'),
        in: headersAtt.indexOf('دخول'),
        out: headersAtt.indexOf('خروج'),
        sFrom: headersAtt.indexOf('ساعية (من الساعة)'),
        sTo: headersAtt.indexOf('ساعية (إلى الساعة)'),
        mFrom: headersAtt.indexOf('مهمة (من الساعة)'),
        mTo: headersAtt.indexOf('مهمة (إلى الساعة)'),
        notes: headersAtt.indexOf('ملاحظات'),
        days: headersAtt.indexOf('عدد الأيام المحتسبة')
      };
      let count = 0;
      attendanceData.forEach(r=>{
        if (!r.length || String(r[idx.code])!==currentUser) return;
        count++;
        const tr = document.createElement('tr');
        tr.innerHTML = `
          <td class="px-4 py-2">${r[idx.date]||''}</td>
          <td class="px-4 py-2">${caseMapping[r[idx.status]]||r[idx.status]||''}</td>
          <td class="px-4 py-2">${r[idx.in]||''}</td>
          <td class="px-4 py-2">${r[idx.out]||''}</td>
          <td class="px-4 py-2">${r[idx.sFrom]||''}</td>
          <td class="px-4 py-2">${r[idx.sTo]||''}</td>
          <td class="px-4 py-2">${r[idx.mFrom]||''}</td>
          <td class="px-4 py-2">${r[idx.mTo]||''}</td>
          <td class="px-4 py-2">${r[idx.notes]||''}</td>
          <td class="px-4 py-2">${r[idx.days]||''}</td>
        `;
        tbody.appendChild(tr);
      });
      document.getElementById('noDataMsg').classList.toggle('hidden', count>0);
    }

    // 7) حفظ PDF
    async function savePDF(){
      const element = document.getElementById('records');
      const canvas  = await html2canvas(element,{ scale:2 });
      const img     = canvas.toDataURL('image/png');
      const pdf     = new jsPDF('l','pt','a4');
      const w       = pdf.internal.pageSize.getWidth();
      const h       = (canvas.height * w)/canvas.width;
      pdf.addImage(img,'PNG',0,20,w,h);
      pdf.save(`attendance_${currentUser}.pdf`);
    }

    function showAbout(){ alert('فكرة وتصميم عمر الماضي - اتصالات دمشق'); }
    function logout(){
      currentUser=null;
      document.getElementById('records').classList.add('hidden');
      document.getElementById('loginSection').classList.remove('hidden');
    }
  </script>
